CREATE DATABASE beauty_hair IF NOT EXISTS;

CREATE TABLE IF NOT EXISTS clients (
id int NOT NULL AUTO_INCREMENT,
first_name varchar(45),
last_name varchar(45),
gender varchar(1),
registration_date DATATIME,
PRIMARY KEY (id)
);

CREATE TABLE IF NOT EXISTS contacts (
id int NOT NULL AUTO_INCREMENT,
value VARCHAR(160),
type INT,
primary TINYINT,
FOREIGN KEY (client_id) REFERENCES clients(id)
);

CREATE TABLE IF NOT EXISTS orders (
id int NOT NULL AUTO_INCREMENT,
start_time DATATIME,
end_time DATATIME,
FOREIGN KEY (service_sale_tips) REFERENCES incomes(id),
FOREIGN KEY (client_id) REFERENCES clients(id)
);

CREATE TABLE IF NOT EXISTS incomes (
id int NOT NULL AUTO_INCREMENT,
name VARCHAR(45),
FOREIGN KEY (good_id) REFERENCES goods(id),
FOREIGN KEY (product_id) REFERENCES store(id)
);

CREATE TABLE IF NOT EXISTS goods (
id int NOT NULL AUTO_INCREMENT,
name VARCHAR(128),
type VARCHAR(10),
price FLOAT
);

CREATE TABLE IF NOT EXISTS store (
id int NOT NULL AUTO_INCREMENT,
product VARCHAR(45),
amount INT
);

CREATE TABLE IF NOT EXISTS expenses (
id int NOT NULL AUTO_INCREMENT,
business_tax FLOAT,
rent FLOAT,
advertising FLOAT,
buying_cosmetics FLOAT,
FOREIGN KEY (product_id) store(id)
)
